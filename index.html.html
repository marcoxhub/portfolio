    Consulta SQL Puxar Quantidade do Estoque WMS e GERENCIAL (TOTVS WINTHOR WMS)

      select PE.codfilial AS FILIAL,PE.codprod AS CODIGO,PT.DESCRICAO,PE.qtest AS WMS,PE.qtestger AS GERENCIAL, PE.qtindeniz AS AVARIA
                    , PE.qtbloqueada AS BLOQUEADO,pe.dtultent as ULTIMA_ENTRADA,pe.dtultsaida AS ULTIMA_SAIDA ,pe.dtultinvent AS ULTIMO_INVENTARIO from pcest PE
                    
                    INNER JOIN PCPRODUT PT
                    ON PE.CODPROD = PT.CODPROD
                    
                     where PE.codfilial between 1 and 2 AND PE.CODPROD = :CODPROD AND PE.CODFILIAL = :CODFILIAL 
             
                    order by PE.codfilial,PE.codprod ;;

    Consulta SQL PEDIDOS E ROTAS (TOTVS WINTHOR WMS)

    SELECT  DISTINCT(PM.NUMOS),PM.NUMPED,PV.NUMVOL,PM.CODFUNCCONF ||'-'|| PER.NOME,PP.ROTA,PM.DTFIMOS FROM PCMOVENDPEND PM
 
                    INNER JOIN PCPRACA PP
                    ON PM.CODPRACA = PP.CODPRACA
                    
                    INNER JOIN PCEMPR PER
                    ON PM.CODFUNCCONF = PER.MATRICULA
                    
                    INNER JOIN PCVOLUMEOS PV
                    ON PM.NUMVOL = PV.NUMVOL
                    
                    
                    
                   
                    WHERE PM.DATA BETWEEN :DTINI AND :DTFIM;

    Consulta SQL PUXAR OS CAIXA FECHADA (TOTVS WINTHOR WMS)

     SELECT DISTINCT PM.NUMPED,(PM.NUMOS),PM.CODPROD,PT.DESCRICAO,PM.QTSEPARADA,PP.ROTA,(PM.NUMVOL),PM.DTFIMOS,PR.NOME FROM PCMOVENDPEND PM
 
                        INNER JOIN PCPRACA PP
                        ON PM.CODPRACA = PP.CODPRACA
                        
                        INNER JOIN PCPRODUT PT
                        ON PM.CODPROD = PT.CODPROD
                        
                        INNER JOIN PCEMPR PR
                        ON PM.CODFUNCOS = PR.MATRICULA
                        
                       
                        WHERE PM.DATA BETWEEN :DTINI AND :DTFIM AND PM.TIPOOS = 20 ;
